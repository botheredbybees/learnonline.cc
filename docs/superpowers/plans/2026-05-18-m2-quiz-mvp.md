# M2 Quiz MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire AQTF units → element-level quizzes → XP/badges, with AI-pre-generated question packs that teachers download and approve before students see them.

**Architecture:** Extend existing FastAPI + SQLAlchemy stack with new quiz tables (user_element_progress, user_answers, question_packs), modify assessments and assessment_questions to be element-scoped with JSONB options, add two new routers (quiz.py, packs.py), a developer-side Ollama generation script, and three new/extended frontend pages (unit.html, quiz.html, admin tabs, dashboard cards).

**Tech Stack:** FastAPI, SQLAlchemy 2.x, Alembic, PostgreSQL (JSONB), Ollama/qwen2.5:14b, Bootstrap 5.3.2, jQuery 3.7.1, vanilla JS

import uuid
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from database import get_db
from auth.auth_bearer import JWTBearer
from auth.auth_handler import get_current_user
import models.tables as models
from models.schemas import (
    UnitQuizStateResponse, ElementStatusSchema,
    QuestionResponse, AnswerSubmission, AnswerResponse,

from services.points import award_points

router = APIRouter(prefix="/api/quiz", tags=["quiz"])


def _get_element_status(db: Session, user_id: int, element_id: int) -> str:
    prog = db.query(models.UserElementProgress).filter_by(
        user_id=user_id, element_id=element_id
    ).first()
    return prog.status if prog else "not_started"


def _grade_mcq(options: dict, answer: dict) -> tuple[bool, str]:
    correct = options.get("correct")
    selected = answer.get("selected")
    explanation = options.get("explanation", "")
    return selected == correct, explanation


def _grade_short_answer(options: dict, answer: dict) -> tuple[bool, str]:
    keywords = [k.lower() for k in options.get("keywords", [])]
    response = answer.get("text", "").lower()
    matched = any(kw in response for kw in keywords)
    return matched, options.get("model_answer", "")


def _grade_ordering(options: dict, answer: dict) -> tuple[bool, str]:
    correct_order = options.get("correct_order", [])
    submitted = answer.get("order", [])
    return submitted == correct_order, "Correct order: " + str(options.get("items", []))


def _check_element_completion(db: Session, user_id: int, element_id: int, session_id: str, assessment_xp: int) -> Optional[int]:
    """Return XP if all questions in element answered correctly this session, else None."""
    assessment = db.query(models.Assessment).filter_by(element_id=element_id).first()
    if not assessment:
        return None

    total_questions = db.query(models.AssessmentQuestion).filter_by(
        assessment_id=assessment.id, is_active=True, review_status="approved"
    ).count()

    session_correct = db.query(models.UserAnswer).filter_by(
        user_id=user_id, session_id=session_id, is_correct=True
    ).join(models.AssessmentQuestion).filter(
        models.AssessmentQuestion.assessment_id == assessment.id
    ).count()

    if session_correct >= total_questions:
        return assessment.xp_awarded if hasattr(assessment, 'xp_awarded') else assessment_xp
    return None


@router.get("/units/{unit_id}/quiz-state", dependencies=[Depends(JWTBearer())])
def get_quiz_state(unit_id: int, db: Session = Depends(get_db), current_user=Depends(get_current_user)):
    unit = db.query(models.Unit).filter_by(id=unit_id).first()
    if not unit:
        raise HTTPException(status_code=404, detail="Unit not found")

    elements = db.query(models.UnitElement).filter_by(unit_id=unit_id).order_by(models.UnitElement.element_num).all()
    element_statuses = []
    for el in elements:
        prog = db.query(models.UserElementProgress).filter_by(
            user_id=current_user.id, element_id=el.id
        ).first()
        element_statuses.append(ElementStatusSchema(
            element_id=el.id,
            element_num=el.element_num,
            element_text=el.element_text,
            status=prog.status if prog else "not_started",
            attempts=prog.attempts if prog else 0,
            xp_awarded=prog.xp_awarded if prog else None,
        ))

    return UnitQuizStateResponse(
        unit_id=unit.id,
        unit_code=unit.code,
        unit_title=unit.title,
        plain_english_description=unit.plain_english_description,
        elements=element_statuses,
    )


@router.get("/elements/{element_id}/questions", dependencies=[Depends(JWTBearer())])
def get_element_questions(element_id: int, db: Session = Depends(get_db)):
    assessment = db.query(models.Assessment).filter_by(element_id=element_id).first()
    if not assessment:
        raise HTTPException(status_code=404, detail="No quiz for this element")

    questions = db.query(models.AssessmentQuestion).filter_by(
        assessment_id=assessment.id,
        is_active=True,
        review_status="approved",
    ).all()

    random.shuffle(questions)
    return [QuestionResponse(
        id=q.id,
        question_text=q.question_text,
        question_type=q.question_type,
        options=q.options or {},
        pc_id=q.pc_id,
    ) for q in questions]


@router.post("/answer", dependencies=[Depends(JWTBearer())])
def submit_answer(
    payload: AnswerSubmission,
    db: Session = Depends(get_db),
    current_user=Depends(get_current_user),
) -> AnswerResponse:
    question = db.query(models.AssessmentQuestion).filter_by(id=payload.question_id).first()
    if not question:
        raise HTTPException(status_code=404, detail="Question not found")

    opts = question.options or {}
    graders = {
        "mcq": _grade_mcq,
        "short_answer": _grade_short_answer,
        "ordering": _grade_ordering,
    }
    grader = graders.get(question.question_type)
    if not grader:
        raise HTTPException(status_code=400, detail=f"Unknown question type: {question.question_type}")

    is_correct, explanation = grader(opts, payload.answer)

    answer_record = models.UserAnswer(
        user_id=current_user.id,
        question_id=payload.question_id,
        session_id=payload.session_id,
        answer=payload.answer,
        is_correct=is_correct,
    )
    db.add(answer_record)
    db.flush()

    assessment = db.query(models.Assessment).filter_by(element_id=payload.element_id).first()
    xp_per_element = assessment.experience_points if assessment else 50

    element_passed = False
    xp_awarded = None
    unit_completed = False
    badge_awarded = None

    xp = _check_element_completion(db, current_user.id, payload.element_id, payload.session_id, xp_per_element)
    if xp is not None:
        prog = db.query(models.UserElementProgress).filter_by(
            user_id=current_user.id, element_id=payload.element_id
        ).first()
        if prog is None or prog.status != "passed":
            if prog is None:
                element = db.query(models.UnitElement).filter_by(id=payload.element_id).first()
                prog = models.UserElementProgress(
                    user_id=current_user.id,
                    element_id=payload.element_id,
                    unit_id=element.unit_id,
                )
                db.add(prog)
            prog.status = "passed"
            prog.xp_awarded = xp
            prog.attempts = (prog.attempts or 0) + 1
            from datetime import datetime, timezone
            prog.passed_at = datetime.now(timezone.utc)
            db.flush()

            award_points(db, current_user.id, xp)
            element_passed = True
            xp_awarded = xp

            element_obj = db.query(models.UnitElement).filter_by(id=payload.element_id).first()
            all_elements = db.query(models.UnitElement).filter_by(unit_id=element_obj.unit_id).all()
            all_element_ids = [e.id for e in all_elements]
            passed_count = db.query(models.UserElementProgress).filter(
                models.UserElementProgress.user_id == current_user.id,
                models.UserElementProgress.element_id.in_(all_element_ids),
                models.UserElementProgress.status == "passed",
            ).count()

            if passed_count == len(all_element_ids):
                unit_completed = True
                unit = db.query(models.Unit).filter_by(id=element_obj.unit_id).first()
                badge = db.query(models.Badge).filter_by(code=unit.code).first()
                if badge is None:

                if badge:

                        user_id=current_user.id, badge_id=badge.id
                    ).first()
                    if not existing:
                        db.add(models.UserBadge(user_id=current_user.id, badge_id=badge.id))
                    badge_awarded = badge.title

    db.commit()

    return AnswerResponse(
        is_correct=is_correct,
        explanation=explanation,
        element_passed=element_passed,
        xp_awarded=xp_awarded,
        unit_completed=unit_completed,
        badge_awarded=badge_awarded,
    )

@router.post("/packs/import", dependencies=[Depends(JWTBearer())])
def import_pack(payload: dict, db: Session = Depends(get_db)) -> PackImportResponse:
    """
    Import a question pack JSON. Upserts unit structure and questions.
    All questions imported with review_status='draft', is_active=False.
    """
    pack_data = payload.get("pack_data")
    if not pack_data:
        raise HTTPException(status_code=400, detail="pack_data required")

    tp_code = pack_data.get("training_package", "UNKNOWN")
    total_questions = 0

    pack_record = models.QuestionPack(
        training_package_code=tp_code,
        source_url=pack_data.get("source_url", ""),
        version=pack_data.get("generated_at", ""),
        status="pending",
    )
    db.add(pack_record)
    db.flush()

    for unit_data in pack_data.get("units", []):
        unit = _upsert_unit(db, unit_data, tp_code)

        for element_data in unit_data.get("elements", []):
            element = _upsert_element(db, element_data, unit)
            assessment = _upsert_assessment(db, element, unit)

            # Map pc_code → pc_id
            pc_map = {}
            for pc_data in element_data.get("performance_criteria", []):
                pc = db.query(models.UnitPerformanceCriteria).filter_by(
                    element_id=element.id, pc_num=pc_data["code"]
                ).first()
                if pc is None:
                    pc = models.UnitPerformanceCriteria(
                        element_id=element.id,
                        unit_id=unit.id,
                        pc_num=pc_data["code"],
                        pc_text=pc_data["text"],
                    )
                    db.add(pc)
                    db.flush()
                pc_map[pc_data["code"]] = pc.id

            for q_data in element_data.get("questions", []):
                q = models.AssessmentQuestion(
                    assessment_id=assessment.id,
                    pc_id=pc_map.get(q_data.get("pc_code")),

                    source=q_data.get("source", "ai_generated"),

    pack_record.question_count = total_questions
    pack_record.status = "imported"
    pack_record.imported_at = datetime.now(timezone.utc)
    db.commit()

    return PackImportResponse(
        pack_id=pack_record.id,
        training_package_code=tp_code,
        unit_count=len(pack_data.get("units", [])),

        status="imported",

#!/usr/bin/env python3
"""
Generate question packs from AQTF data using a local Ollama instance.

Usage:
    python backend/scripts/generate_questions.py --package MSF --model qwen2.5:14b
    python backend/scripts/generate_questions.py --package MSF --model qwen2.5:14b --unit MSFFI2001

Outputs: questions-{PACKAGE}.json in the current directory.

Requirements:
    pip install requests sqlalchemy psycopg2-binary python-dotenv
    Ollama running locally at http://localhost:11434
"""
import argparse
import json
import os
import sys
import time
from datetime import date
from pathlib import Path

import requests
from dotenv import load_dotenv
from sqlalchemy import create_engine, text

load_dotenv()

DATABASE_URL = os.environ.get("DATABASE_URL")
if not DATABASE_URL:
    print("ERROR: DATABASE_URL not set in .env", file=sys.stderr)
    sys.exit(1)

OLLAMA_URL = os.environ.get("OLLAMA_URL", "http://localhost:11434")

MCQ_PROMPT = """You are an expert vocational education assessor for Australian training packages.

Generate a multiple choice question for this performance criterion:
  PC: {pc_text}

Context (unit plain English description):
  {description}

Knowledge Evidence:
  {knowledge_evidence}

Return ONLY valid JSON in this exact format:
{{
  "question_text": "...",
  "options": {{
    "choices": ["option A", "option B", "option C", "option D"],
    "correct": 0,
    "explanation": "Brief explanation of the correct answer."
  }}
}}
"""

SHORT_ANSWER_PROMPT = """You are an expert vocational education assessor for Australian training packages.

Generate a short answer question for this knowledge evidence item:
  {knowledge_evidence}

Context (unit plain English description):
  {description}

Return ONLY valid JSON in this exact format:
{{
  "question_text": "...",
  "options": {{
    "keywords": ["keyword1", "keyword2", "keyword3"],
    "model_answer": "A complete model answer the student should approximate."
  }}
}}
"""

ORDERING_PROMPT = """You are an expert vocational education assessor for Australian training packages.

Generate a sequencing/ordering question for this performance criterion involving a procedure:
  PC: {pc_text}

Context:
  {description}

Return ONLY valid JSON in this exact format:
{{
  "question_text": "Place these steps in the correct order:",
  "options": {{
    "items": ["step A", "step B", "step C", "step D"],
    "correct_order": [2, 0, 3, 1]
  }}
}}
The correct_order array contains the 0-based indices that form the correct sequence.
"""


def call_ollama(model: str, prompt: str) -> dict:
    """Call Ollama generate endpoint and return parsed JSON from response."""
    resp = requests.post(
        f"{OLLAMA_URL}/api/generate",
        json={"model": model, "prompt": prompt, "stream": False},
        timeout=120,
    )
    resp.raise_for_status()
    raw = resp.json().get("response", "")
    # Extract JSON block from response
    start = raw.find("{")
    end = raw.rfind("}") + 1
    if start == -1 or end == 0:
        raise ValueError(f"No JSON in Ollama response: {raw[:200]}")
    return json.loads(raw[start:end])


def fetch_units(engine, package_code: str, unit_code: str = None) -> list:
    """Fetch units for a training package from the lol_units table."""
    where = "WHERE code LIKE :prefix"
    params = {"prefix": f"{package_code}%"}
    if unit_code:
        where = "WHERE code = :unit_code"
        params = {"unit_code": unit_code}

    with engine.connect() as conn:
        rows = conn.execute(
            text(f"SELECT code, title, plain_english_description FROM lol_units {where} ORDER BY code"),
            params,
        ).fetchall()
    return [{"code": r[0], "title": r[1], "plain_english_description": r[2]} for r in rows]


def fetch_unit_elements(engine, unit_code: str) -> list:
    """Fetch elements and performance criteria for a unit from the learnonline DB."""
    with engine.connect() as conn:
        unit_row = conn.execute(
            text("SELECT id FROM units WHERE code = :code"),
            {"code": unit_code},
        ).fetchone()
        if not unit_row:
            return []
        unit_id = unit_row[0]

        elements = conn.execute(
            text("SELECT id, element_num, element_text FROM unit_elements WHERE unit_id = :uid ORDER BY element_num"),
            {"uid": unit_id},
        ).fetchall()

        result = []
        for el_id, el_num, el_text in elements:
            pcs = conn.execute(
                text("SELECT id, pc_num, pc_text FROM unit_performance_criteria WHERE element_id = :eid ORDER BY pc_num"),
                {"eid": el_id},
            ).fetchall()
            kas = conn.execute(
                text("SELECT critical_aspect FROM unit_critical_aspects WHERE unit_id = :uid"),
                {"uid": unit_id},
            ).fetchall()

            result.append({
                "code": el_num,
                "title": el_text,
                "performance_criteria": [{"code": pc[1], "text": pc[2]} for pc in pcs],
                "critical_aspects": [{"text": ka[0]} for ka in kas],
            })
    return result


def generate_questions_for_element(model: str, element: dict, description: str) -> list:
    """Generate MCQ + short answer for each PC; ordering only for first PC."""
    questions = []
    knowledge_text = " ".join(ka["text"] for ka in element.get("critical_aspects", []))

    for i, pc in enumerate(element.get("performance_criteria", [])):
        # MCQ for every PC
        try:
            mcq = call_ollama(model, MCQ_PROMPT.format(
                pc_text=pc["text"],
                description=description or "",
                knowledge_evidence=knowledge_text,
            ))
            questions.append({
                "pc_code": pc["code"],
                "question_type": "mcq",
                "question_text": mcq["question_text"],
                "options": mcq["options"],
                "source": "ai_generated",
                "review_status": "draft",
            })
        except Exception as e:
            print(f"    MCQ failed for {pc['code']}: {e}", file=sys.stderr)

        # Short answer for every PC
        try:
            sa = call_ollama(model, SHORT_ANSWER_PROMPT.format(
                knowledge_evidence=knowledge_text or pc["text"],
                description=description or "",
            ))
            questions.append({
                "pc_code": pc["code"],
                "question_type": "short_answer",
                "question_text": sa["question_text"],
                "options": sa["options"],
                "source": "ai_generated",
                "review_status": "draft",
            })
        except Exception as e:
            print(f"    Short answer failed for {pc['code']}: {e}", file=sys.stderr)

        # Ordering only for first PC (procedural context)
        if i == 0 and len(element.get("performance_criteria", [])) >= 3:
            try:
                ordering = call_ollama(model, ORDERING_PROMPT.format(
                    pc_text=pc["text"],
                    description=description or "",
                ))
                questions.append({
                    "pc_code": pc["code"],
                    "question_type": "ordering",
                    "question_text": ordering["question_text"],
                    "options": ordering["options"],
                    "source": "ai_generated",
                    "review_status": "draft",
                })
            except Exception as e:
                print(f"    Ordering failed for {pc['code']}: {e}", file=sys.stderr)

    return questions


def main():

    parser.add_argument("--package", required=True, help="Training package code, e.g. MSF")

    parser.add_argument("--unit", default=None, help="Limit to a single unit code")
    parser.add_argument("--output", default=None, help="Output file path (default: questions-{PACKAGE}.json)")
    args = parser.parse_args()

    output_path = args.output or f"questions-{args.package}.json"

    engine = create_engine(DATABASE_URL)

    print(f"Fetching units for package {args.package}...")
    units = fetch_units(engine, args.package, args.unit)
    if not units:
        print(f"No units found for package {args.package}", file=sys.stderr)
        sys.exit(1)
    print(f"Found {len(units)} units.")

    pack = {
        "training_package": args.package,
        "training_package_title": "",
        "generated_at": str(date.today()),
        "model": args.model,
        "qualifications": [],
        "units": [],
    }

    for i, unit_data in enumerate(units, 1):
        code = unit_data["code"]
        print(f"[{i}/{len(units)}] {code}: {unit_data['title']}")

        elements = fetch_unit_elements(engine, code)
        if not elements:
            print(f"  No elements found, skipping.")
            continue

        unit_entry = {
            "code": code,
            "title": unit_data["title"],
            "description": "",
            "plain_english_description": unit_data.get("plain_english_description", ""),
            "elements": [],
        }

        for el in elements:
            print(f"  Element {el['code']}: generating questions...")
            questions = generate_questions_for_element(
                args.model, el, unit_data.get("plain_english_description", "")
            )
            el["questions"] = questions
            unit_entry["elements"].append(el)
            print(f"    Generated {len(questions)} questions.")
            time.sleep(0.2)  # brief pause between Ollama calls

        pack["units"].append(unit_entry)

    with open(output_path, "w") as f:
        json.dump(pack, f, indent=2)

    total_q = sum(len(q) for u in pack["units"] for e in u["elements"] for q in [e.get("questions", [])])
    print(f"\nDone. {len(pack['units'])} units, {total_q} questions → {output_path}")


if __name__ == "__main__":
    main()

                                    <button class="btn btn-sm btn-success" onclick="approveQuestion(${q.id})">Approve</button>

function approveQuestion(id) {

        contentType: 'application/json',
        headers: { Authorization: 'Bearer ' + localStorage.getItem('token') },

function bulkApprove() {
    $.ajax({
        url: '/api/admin/questions?review_status=draft&limit=500',
        headers: { Authorization: 'Bearer ' + localStorage.getItem('token') },
        success(questions) {
            const ids = questions.map(q => q.id);
            let done = 0;
            if (!ids.length) { alert('No draft questions.'); return; }
            ids.forEach(id => {
                $.ajax({
                    url: '/api/admin/questions/' + id,
                    method: 'PATCH',
                    contentType: 'application/json',
                    headers: { Authorization: 'Bearer ' + localStorage.getItem('token') },
                    data: JSON.stringify({ review_status: 'approved', is_active: true }),
                    success() { done++; if (done === ids.length) { alert(`Approved ${done} questions.`); loadQuestions(); } }
                });
            });
        }
    });
}

```javascript
function loadMyUnits() {
    const token = localStorage.getItem('token');
    if (!token) return;

    $.ajax({
        url: '/api/units/?limit=200',
        headers: { Authorization: 'Bearer ' + token },
        success(units) {
            const section = $('#my-units-section');
            const list = $('#unit-cards-list');
            list.empty();
            let hasAny = false;

            let remaining = units.length;
            if (!remaining) return;

            units.forEach(unit => {
                $.ajax({
                    url: `/api/quiz/units/${unit.id}/quiz-state`,
                    headers: { Authorization: 'Bearer ' + token },
                    success(state) {
                        const started = state.elements.some(e => e.status !== 'not_started');
                        if (started) {
                            hasAny = true;
                            const passed = state.elements.filter(e => e.status === 'passed').length;
                            const total = state.elements.length;
                            const pct = total ? Math.round((passed / total) * 100) : 0;
                            const xp = state.elements.reduce((sum, e) => sum + (e.xp_awarded || 0), 0);
                            list.append(`
                                <div class="unit-progress-card" onclick="location.href='/unit.html?id=${unit.id}'">
                                    <div class="unit-code">${state.unit_code}</div>
                                    <div class="unit-title">${state.unit_title}</div>
                                    <div class="unit-progress-bar-outer">
                                        <div class="unit-progress-bar-inner" style="width:${pct}%"></div>
                                    </div>
                                    <div class="unit-xp-total">${passed}/${total} elements · ${xp} XP earned</div>
                                </div>
                            `);
                            section.show();
                        }
                    },
                    complete() {
                        remaining--;
                        if (remaining === 0 && !hasAny) section.hide();
                    }
                });
            });
        }
    });
}

$(document).ready(function() {
    loadMyUnits();
});
```
